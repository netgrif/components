import {Component, Injector} from '@angular/core';
import {AbstractTabViewComponent, LoggerService, ViewService} from '@netgrif/components-core';

/**
 * Component that renders a tab view.
 *
 * See {@link TabView} for the class that holds the logic for this view.
 */
@Component({
    selector: 'nc-tab-view',
    templateUrl: './tab-view.component.html',
    styleUrls: ['./tab-view.component.scss']
})
export class TabViewComponent extends AbstractTabViewComponent {

    constructor(protected _viewService: ViewService, protected _logger: LoggerService, injector: Injector) {
        super(_viewService, _logger, injector);
    }
}

<mat-tab-group [selectedIndex]="tabView.selectedIndex.value"

    <mat-tab *ngFor="let tab of tabView.openedTabs; let i=index">
        <ng-template mat-tab-label>
            <div fxLayout="row" fxLayoutAlign="start center">
                <div fxFlex fxLayout="row" fxLayoutAlign="start center">
                    <mat-icon class="icon-margin-right"
                              *ngIf="tab.getIcon() !== undefined">{{tab.getIcon$()|async}}</mat-icon>
                    <span class="custom-badge" [matBadge]="(badgeCount(tab) | async)"

                </div>
                <button mat-icon-button *ngIf="tab.canBeClosed" (click)="tabView.closeTabIndex(i)">
                    <mat-icon>clear</mat-icon>
                </button>
            </div>
        </ng-template>

            <ng-template [cdkPortalOutlet]="tab.portal" class="full-height"></ng-template>
        </ng-template>
    </mat-tab>
</mat-tab-group>
